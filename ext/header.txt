<script>
    // Initialize tooltips
    const tooltipTriggerList = [].slice.call(document.querySelectorAll('[data-bs-toggle="tooltip"]'))
    const tooltipList = tooltipTriggerList.map(function (tooltipTriggerEl) {
        return new bootstrap.Tooltip(tooltipTriggerEl)
    })
    
    // Handle authentication messages
    document.addEventListener('DOMContentLoaded', function() {
        // SignUp message handling
        const signupReport = document.getElementById('user_signup_report').value;
        if(signupReport && signupReport !== '') {
            const messageBox = new bootstrap.Modal(document.getElementById('message_box'));
            const messageTitle = document.getElementById('message_title');
            const messageContent = document.getElementById('message');
            
            messageTitle.innerText = 'Sign Up Status';
            messageContent.innerText = signupReport === 'fail' ? 
                'Registration failed. Please try again with different credentials.' :
                'Registration successful! Please sign in with your credentials.';
            
            messageBox.show();
        }
        
        // SignIn message handling
        const signinReport = document.getElementById('user_signin_report').value;
        if(signinReport && signinReport === 'fail') {
            const messageBox = new bootstrap.Modal(document.getElementById('message_box'));
            const messageTitle = document.getElementById('message_title');
            const messageContent = document.getElementById('message');
            
            messageTitle.innerText = 'Sign In Status';
            messageContent.innerText = 'Sign in failed. Please check your credentials and try again.';
            
            messageBox.show();
        }
        
        // Form validation
        const forms = document.querySelectorAll('.needs-validation');
        Array.from(forms).forEach(form => {
            form.addEventListener('submit', event => {
                if (!form.checkValidity()) {
                    event.preventDefault();
                    event.stopPropagation();
                }
                form.classList.add('was-validated');
            }, false);
        });
    });
</script>
